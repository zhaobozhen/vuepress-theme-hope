# As Easy as 1, 2, 3

# find a proper place to clone this project template
git clone git@github.com:Mister-Hope/vuepress-theme-hope-template.git

# enter the program folder
cd vuepressTemplate

# install dependency
npm i

# start writing
npm run serve

# build to static files
npm run build
